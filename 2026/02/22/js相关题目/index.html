生成随机数

  1. 生成 [min, max) 之间的随机整数（包含min，不包含max）
    function getRandomIntInRange(min, max) {
      return Math.floor(Math.random() * (max - min)) + min;
    }

  2. 生成 [min, max] 之间的随机整数（包含min和max）
    function getRandomIntInclusive(min, max) {
      return Math.floor(Math.random() * (max - min + 1)) + min;
    }

// 示例：生成1到10（含）的随机整数
const diceRoll = Math.floor(Math.random() * 10) + 1;
console.log(diceRoll); // 结果可能是 1, 2, …, 10

Math.random()：返回一个介于 0(包含) 和 1(不包含) 之间的伪随机浮点数。
Math.floor()：向下取整，舍弃小数部分。
因此 **Math.floor(Math.random() * n)**能将 [0, 1)的随机数映射到 [0, n-1]的整数范围。

  1. 生成指定数目和范围的不重复随机数
    const getRandomNum = function(min,max,countNum){
      var arr = [];
      // 使用Set来确保不重复
      let set = new Set();

 
  // 当Set中的元素数量小于所需数量时继续生成
  while(set.size < countNum) {
  // 生成 min~max 范围的随机整数
  let randomNum = Math.floor(Math.random() * (max - min + 1)) + min;
  set.add(randomNum);
  }
 
  // 将Set转换为数组
  arr = Array.from(set);
  return arr;
}
module.exports = getRandomNum; //请勿删除

步骤（从后向前）：
设数组长度为 n。
从 i = n-1开始递减至 1：
生成随机索引 j，满足 0 ≤ j ≤ i。
交换 arr[i]和 arr[j]。
完成打乱。
function fisherYatesShuffle(arr) {
  const n = arr.length;
  for (let i = n - 1; i > 0; i–) {
  const j = Math.floor(Math.random() * (i + 1)); // 随机选择 0 到 i 的索引
  [arr[i], arr[j]] = [arr[j], arr[i]]; // 交换元素
  }
  return arr;
}

// 使用示例
const cards = [1, 2, 3, 4, 5];
console.log(fisherYatesShuffle([…cards])); // 每次运行得到随机顺序

验证密码强度

在密码框中输入密码。
点击“验证”按钮。
输出密码强度等级：低、中、高。
而密码强度的等级在上图中也给出了划分。具体如下：
低：
密码必须大于 8 个字符。
中（在满足低强度要求的前提下，需要满足以下需求）：
至少需要一个小写字母。
至少需要一个数字。
高（在满足中强度要求的前提下，需要满足以下需求）：
至少需要一个大写字母。
至少需要一个（除数字和字母外的）特殊字符。
需要注意的是，当密码小于等于 8 个字符的情况下点击“验证”按钮后，需要在强度等级中输出“无效”的结果。

document.querySelector(‘button’).addEventListener(‘click’,()=>{
  const pwd=document.querySelector(‘input’).value
 
  // 首先检查长度（题目要求：小于等于8个字符为无效）
  if (pwd.length <= 8) {
  document.querySelector(‘.re’).innerHTML = ‘无效’;
  return;
  }

  // 检查各种字符类型
  const hasLowCase = /[a-z]/.test(pwd); // 小写字母
  const hasUpperCase = /[A-Z]/.test(pwd); // 大写字母
  const hasNumber = /\d/.test(pwd); // 数字
  const hasSpecialCase = /[^0-9a-zA-Z]/.test(pwd); // 特殊字符
 
  // 强度判断（从高到低检查）
  if (hasLowCase && hasNumber && hasUpperCase && hasSpecialCase) {
  // 高强度：满足所有条件
  document.querySelector(‘.re’).innerHTML = ‘高’;
  } else if (hasLowCase && hasNumber) {
  // 中强度：至少小写字母和数字
  document.querySelector(‘.re’).innerHTML = ‘中’;
  } else {
  // 低强度：仅满足长度>8
  document.querySelector(‘.re’).innerHTML = ‘低’;
  }
})

防抖

登录、发短信等按钮避免用户点击太快，以致于发送了多次请求。
调整浏览器窗口大小时，resize 次数过于频繁，造成计算过多。
文本编辑器实时保存。
上面这些场景在实际交互中会不断地调用绑定在事件（如：浏览器的 resize、scroll、keypress、mousemove 等）上的回调函数，极大地浪费资源，降低前端性能。为了优化体验，需要对这类事件进行调用次数的限制，对此我们就可以采用防抖（debounce）的方式来减少调用频率。

来看一下防抖的定义：n 秒后再执行某个函数，若该函数在 n 秒内被重复触发，则重新计时。
防抖函数的核心思想是：在短时间内多次触发同一事件时，只执行最后一次，适用于输入框搜索、窗口resize等场景。存储定时器ID：使用闭包保存 timeout
清除旧定时器：每次调用都清除之前的等待
设置新定时器：延迟执行原始函数
保持上下文：使用 apply()确保 this正确

function debounce(fn,delay=0){
let timeout=null;
return function(…args){
clearTimeout(timeout)
timeout=setTimeout(()=>{
fn.apply(this,args)
},delay)
}
}

fn:需要防抖处理的原始函数
delay:防抖延迟时间
timeout用于保存 setTimeout返回的计时器ID
由于闭包，每次调用返回的函数都能访问这个变量
这个变量是防抖功能的核心
return function(…args) {
返回一个新的函数（防抖版本）
使用扩展运算符 …args接收任意数量的参数
这个函数就是用户实际调用的防抖函数
关键步骤：每次调用防抖函数时，都清除之前设置的定时器
这是实现”防抖”的核心：取消上一次未执行的调用
clearTimeout(null)是安全的，不会报错
创建一个新的定时器，延迟 delay毫秒后执行
fn.apply(this, args):
apply()方法调用原始函数
this保持了调用防抖函数时的上下文
args是传递给防抖函数的参数

在文件 age.js 中，有一个 person 对象，当它的 age 属性值改变时需要对新值做判断，将其永远控制在 0 - 150 之间。具体需求如下：
如果新属性值在 0 -150 之间（包含 0 和 150），则直接更新。
如果新属性值小于 0，则属性值更新为 0。
如果新属性值大于 150，则属性值更新为 150。
// 请不要更改这个对象里面的内容
let person = {
age: 0,
};

Proxy

Proxy是 ES6 引入的新特性，用于创建一个对象的代理，从而可以拦截和自定义对象的基本操作（如属性读取、赋值、函数调用等）。
基本语法：
const proxy = new Proxy(target, handler);
target：要代理的目标对象
handler：包含拦截方法的对象（称为”陷阱”或”处理器”）

二、代码解析

  1. 目标对象
    let person = {
    age: 0,
    };
    这是一个普通的对象，有初始的 age属性值为 0。

  2. 创建 Proxy 代理
    const personProxy = new Proxy(person, {
    // handler 对象包含拦截方法
    set(target, property, value) {
    // 拦截属性赋值操作
    },

get(target, property) {
// 拦截属性读取操作
}
});
personProxy是 person的代理对象，对 personProxy的操作会被 handler 拦截。
3. 拦截 set操作（属性赋值）
set(target, property, value) {
// property: 要设置的属性名
// value: 要设置的值

if (property === ‘age’) {
// 验证逻辑
if (value < 0) {
target[property] = 0; // 小于0 → 设为0
} else if (value > 150) {
target[property] = 150; // 大于150 → 设为150
} else {
target[property] = value; // 0-150之间 → 正常赋值
}
return true; // 必须返回 true 表示设置成功
}

// 对于其他属性，直接设置
target[property] = value;
return true;
}

当对 personProxy.age赋值时触发
对 age 属性进行验证和限制
必须返回 true表示赋值成功，返回 false会抛出 TypeError

  1. 拦截 get操作（属性读取）
    get(target, property) {
    return target[property];
    }
    当读取 personProxy.age时触发
    直接返回目标对象的对应属性值
    这里没有特殊处理，只是透传
  2. 导出代理对象
    module.exports = personProxy; // 注意：这里导出的是代理对象
    导出的是代理对象，而不是原始对象。
    三、如何使用
    // 导入的是代理对象
    const person = require(‘./person.js’);

// 1. 正常赋值（在0-150之间）
